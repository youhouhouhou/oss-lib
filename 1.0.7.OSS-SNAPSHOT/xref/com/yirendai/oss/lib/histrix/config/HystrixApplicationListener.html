1   package com.yirendai.oss.lib.histrix.config;
2   
3   import com.netflix.hystrix.strategy.properties.HystrixPropertiesFactory;
4   
5   import lombok.extern.slf4j.Slf4j;
6   
7   import org.springframework.cloud.context.environment.EnvironmentChangeEvent;
8   import org.springframework.context.ApplicationListener;
9   import org.springframework.stereotype.Component;
10  
11  /**
12   * 监听配置变化，判断是否hystrix属性，是则刷新hystrix属性.
13   * 2016/12/8 yanzhang153
14   */
15  @Slf4j
16  @Component
17  public class HystrixApplicationListener implements ApplicationListener<EnvironmentChangeEvent> {
18    @Override
19    public void onApplicationEvent(EnvironmentChangeEvent event) {
20      for (String key : event.getKeys()) {
21        if (key.startsWith("hystrix")) {
22          HystrixPropertiesFactory.reset(); // 清理当前默认配置,下次请求会自动加载最新配置
23          log.info("hystrix properties has changed, event:{}", event.toString());
24          break;
25        }
26      }
27    }
28  }
