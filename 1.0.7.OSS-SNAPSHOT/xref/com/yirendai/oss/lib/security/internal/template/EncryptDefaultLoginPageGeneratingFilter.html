58    public EncryptDefaultLoginPageGeneratingFilter(final AbstractAuthenticationProcessingFilter filter) {

73    private void childInit( //
74      final UsernamePasswordAuthenticationFilter authFilter, //
75      final AbstractAuthenticationProcessingFilter openIdFilter //
76    ) {
77      this.loginPageUrl = this.loginPageUrl != null ? this.loginPageUrl : DEFAULT_LOGIN_PAGE_URL;
78      this.logoutSuccessUrl = this.loginPageUrl + "?logout";
79      this.failureUrl = this.loginPageUrl + "?" + ERROR_PARAMETER_NAME;
80      if (authFilter != null) {
81        this.formLoginEnabled = true;
82        this.usernameParameter = authFilter.getUsernameParameter();
83        this.passwordParameter = authFilter.getPasswordParameter();
84  
85        if (authFilter.getRememberMeServices() instanceof AbstractRememberMeServices) {
86          this.rememberMeParameter = ((AbstractRememberMeServices) authFilter
87            .getRememberMeServices()).getParameter();
88        }
89      }
90  
91      if (openIdFilter != null) {
92        this.openIdEnabled = true;
93        this.openIDusernameParameter = "openid_identifier";
94  
95        if (openIdFilter.getRememberMeServices() instanceof AbstractRememberMeServices) {
96          this.openIDrememberMeParameter = ((AbstractRememberMeServices) openIdFilter
97            .getRememberMeServices()).getParameter();
98        }
99      }
100 
101     this.jsencrypt = this.getJsencrypt();
102   }

181   @Override
182   public void doFilter( //
183     final ServletRequest req, //
184     final ServletResponse res, //
185     final FilterChain chain //
186   ) throws IOException, ServletException {
187     final HttpServletRequest request = (HttpServletRequest) req;
188     final HttpServletResponse response = (HttpServletResponse) res;
189 
190     boolean loginError = childIsErrorPage(request);
191     boolean logoutSuccess = childIsLogoutSuccess(request);
192     if (childIsLoginUrlRequest(request) || loginError || logoutSuccess) {
193       String loginPageHtml = childGenerateLoginPageHtml(request, loginError,
194         logoutSuccess);
195       response.setContentType("text/html;charset=UTF-8");
196       response.setContentLength(loginPageHtml.length());
197       response.getWriter().write(loginPageHtml);
198 
199       return;
200     }
201 
202     chain.doFilter(request, response);
203   }

205   private String childGenerateLoginPageHtml(HttpServletRequest request, boolean loginError,
206     boolean logoutSuccess) {
207     String errorMsg = "none";
208 
209     if (loginError) {
210       HttpSession session = request.getSession(false);
211 
212       if (session != null) {
213         AuthenticationException ex = (AuthenticationException) session
214           .getAttribute(WebAttributes.AUTHENTICATION_EXCEPTION);
215         errorMsg = ex != null ? ex.getMessage() : "none";
216       }
217     }
218 
219     StringBuilder sb = new StringBuilder();
220 
221     sb.append("<html>") //
222       .append("<head>") //
223       .append("<title>Login Page</title>") //
224       .append("</head>");
225 
226     if (this.formLoginEnabled) {
227       sb.append("<body onload='document.f.").append(this.usernameParameter)
228         .append(".focus();'>\n");
229     }
230 
231     if (loginError) {
232       sb.append("<p><font color='red'>Your login attempt was not successful, try again.<br/><br/>Reason: ");
233       sb.append(errorMsg);
234       sb.append("</font></p>");
235     }
236 
237     if (logoutSuccess) {
238       sb.append("<p><font color='green'>You have been logged out</font></p>");
239     }
240 
241     if (this.formLoginEnabled) {
242       final String redirectParam = request.getParameter(PARAM_REDIRECT);
243       final String baseUri = request.getContextPath() + this.authenticationUrl;
244       final String action = isNotBlank(redirectParam) ?  //
245         baseUri + "?" + PARAM_REDIRECT + "=" + CodecUtils.urlEncode(redirectParam) : baseUri;
246       sb.append("<h3>Login with Username and Password.</h3>");
247       sb.append("<form name='f' action='").append(action) //
248         .append("' method='POST' onsubmit='return onSubmit();'>\n");
249       sb.append("<table>\n");
250       sb.append("    <tr><td>User:</td><td><input type='text' name='");
251       sb.append(this.usernameParameter).append("' value='").append("'></td></tr>\n");
252       sb.append("    <tr><td>Password:</td><td><input type='password' name='")
253         .append(this.passwordParameter).append("'/></td></tr>\n");
254 
255       if (this.rememberMeParameter != null) {
256         sb.append("    <tr><td><input type='checkbox' name='")
257           .append(this.rememberMeParameter)
258           .append("'/></td><td>Remember me on this computer.</td></tr>\n");
259       }
260 
261       sb.append("    <tr><td colspan='2'>\n")
262         .append("        <input onclick=\"onClick()\" name=\"submit\" type=\"submit\" value=\"Login\"/>\n")
263         .append("    </td></tr>\n");
264       childRenderHiddenInputs(sb, request);
265       sb.append("</table>\n");
266       sb.append("</form>");
267     }
268 
269     if (this.openIdEnabled) {
270       sb.append("<h3>Login with OpenID Identity</h3>");
271       sb.append("<form name='oidf' action='").append(request.getContextPath())
272         .append(this.openIDauthenticationUrl).append("' method='POST'>\n");
273       sb.append("<table>\n");
274       sb.append("    <tr><td>Identity:</td><td><input type='text' size='30' name='");
275       sb.append(this.openIDusernameParameter).append("'/></td></tr>\n");
276 
277       if (this.openIDrememberMeParameter != null) {
278         sb.append("    <tr><td><input type='checkbox' name='")
279           .append(this.openIDrememberMeParameter)
280           .append("'></td><td>Remember me on this computer.</td></tr>\n");
281       }
282 
283       sb.append("    <tr><td colspan='2'>\n")
284         .append("        <input name=\"submit\" type=\"submit\" value=\"Login\" />\n")
285         .append("    </td></tr>\n");
286       sb.append("</table>\n");
287       childRenderHiddenInputs(sb, request);
288       sb.append("</form>");
289     }
290 
291     // RSA password encrypt
292     if (this.jsencrypt != null) {
293       sb.append("<script type='text/javascript'>\n").append(this.jsencrypt).append("\n</script>\n");
294     } else {
295       sb.append("<script type='text/javascript' src='js/jsencrypt-2.3.1.min.js'></script>"); //
296     }
297     sb.append("<script type='text/javascript'>\n") //
298       .append("    function onClick() {\n") //
299       .append("        var input = document.getElementsByName(\"").append(this.passwordParameter).append("\")[0];") //
300       .append("\n") //
301       .append("        var plaintext = input.value;\n") //
302       .append("        if (! plaintext) {return false;}\n") //
303       .append("        if (plaintext.startsWith('").append(ENCRYPTED_FIELD_PREFIX).append("')) {return true;}\n\n") //
304       .append("        console.log('plaintext:' + plaintext);\n") //
305       .append("        var rsaPublicKey='").append(isNotBlank(this.rsaPublicKey) ? this.rsaPublicKey : "") //
306       .append("';\n") //
307       .append("        if (! rsaPublicKey) {return true;}\n") //
308       .append("        var encrypt = new JSEncrypt();\n") //
309       .append("        encrypt.setPublicKey(rsaPublicKey);\n") //
310       .append("        var encrypted = encrypt.encrypt(plaintext);\n") //
311       .append("        var result = '").append(ENCRYPTED_FIELD_PREFIX).append("' + encrypted;\n") //
312       .append("        input.value = result;\n")
313       .append("        console.log('result:' + input.value);\n") //
314       .append("        if (encrypted) {return true;} else {return false;}\n") //
315       .append("    }\n") //
316       .append("    function onSubmit() {return true;}\n") //
317       .append("</script>\n");
318     sb.append("</body></html>");
319 
320     return sb.toString();
321   }

336   private boolean childIsLoginUrlRequest(HttpServletRequest request) {
337     final String url = request.getQueryString() != null ? //

339     return childMatches(request, url);

342   private boolean childIsErrorPage(HttpServletRequest request) {
343     return childMatches(request, this.failureUrl);

346   private boolean childMatches(final HttpServletRequest request, final String url) {
347     if (!"GET".equals(request.getMethod()) || url == null) {
348       return false;
349     }
350     String uri = request.getRequestURI();
351     int pathParamIndex = uri.indexOf(';');
352 
353     if (pathParamIndex > 0) {
354       // strip everything after the first semi-colon
355       uri = uri.substring(0, pathParamIndex);
356     }
357 
358     if (request.getQueryString() != null) {
359       uri += "?" + request.getQueryString();
360     }
361 
362     if ("".equals(request.getContextPath())) {
363       return uri.equals(url);
364     }
365 
366     final String urlWithContextPath = request.getContextPath() + url;
367     return uri.equals(urlWithContextPath);
368   }
