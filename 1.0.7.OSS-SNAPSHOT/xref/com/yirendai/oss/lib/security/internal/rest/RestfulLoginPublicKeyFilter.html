40  @Slf4j
41  public class RestfulLoginPublicKeyFilter extends GenericFilterBean {
42  
43    private final KeyExpression key;
44  
45    private RequestMatcher requestMatcher;
46  
47    @Autowired
48    @Setter
49    private RestfulExceptionHandler exceptionHandler;
50    @Autowired
51    @Setter
52    private DefaultHttpEntityMethodProcessor httpEntityMethodProcessor;
53  
54    public RestfulLoginPublicKeyFilter(final KeyExpression publicKey) {
55      super();
56      this.key = publicKey;
57    }
58  
59    public KeyExpression getKey() {
60      return this.key;
61    }
62  
63    @Override
64    public void doFilter( //
65      final ServletRequest req, //
66      final ServletResponse res, //
67      final FilterChain chain //
68    ) throws IOException, ServletException {
69      final HttpServletRequest request = (HttpServletRequest) req;
70      final HttpServletResponse response = (HttpServletResponse) res;
71  
72      if (!this.isKeyRequest(request, response)) {
73        chain.doFilter(request, response);
74      } else {
75        final HttpEntity<KeyExpression> returnValue = new HttpEntity<>(this.getKey());
76        final Method method = ClassUtils.getMethod(RestfulLoginPublicKeyFilter.class, "getKey");
77        final MethodParameter returnTypeMethodParam = new MethodParameter(method, -1);
78        final MethodParameter returnType = new MethodParameter(returnTypeMethodParam);
79        final ModelAndViewContainer mavContainer = new ModelAndViewContainer();
80        final NativeWebRequest webRequest = new ServletWebRequest(request, response);
81        try {
82          checkState(isNotBlank(this.key.getValue()), "The key you request is not present.");
83          this.httpEntityMethodProcessor.handleReturnValue(returnValue, returnType, mavContainer, webRequest);
84        } catch (final Exception ex) {
85          if (log.isWarnEnabled()) {
86            log.warn("Failed to process. Output a error response: {}", returnValue, ex);
87          }
88          this.exceptionHandler.resolveAndHandle(request, response, ex);
89        }
90      }
91    }
92  
93    @SuppressWarnings({"squid:S1172"})
94    protected boolean isKeyRequest( //
95      final HttpServletRequest request, //
96      final HttpServletResponse response //
97    ) {
98      return this.requestMatcher.matches(request);
99    }
100

103     this.requestMatcher = requestMatcher;
104   }
