13  import org.springframework.security.core.AuthenticationException;
14  import org.springframework.security.web.authentication.preauth.PreAuthenticatedAuthenticationToken;
15  
16  @Slf4j
17  public class PreAuthTokenAuthenticationProvider implements AuthenticationProvider {
18  
19    private final Boolean bypass;
20  
21    public PreAuthTokenAuthenticationProvider(final Boolean bypass) {
22      this.bypass = bypass;
23    }
24  
25    @Override
26    public Authentication authenticate(final Authentication authentication) throws AuthenticationException {
27      final Authentication result;
28      if (authentication != null && this.supports(authentication.getClass())) {
29        final PreAuthenticatedAuthenticationToken token = (PreAuthenticatedAuthenticationToken) authentication;
30        final Object principal = token.getPrincipal();
31        final Object credentials = token.getCredentials();
32  
33        if (this.bypass || PERMITED.equals(credentials)) {
34          result = AuthenticationTokens.authenticated(token);
35        } else {
36          if (!GenericUser.isGenericUser(principal)) {
37            if (log.isDebugEnabled()) {
38              log.debug("not containsUserInfo");
39            }
40            throw new BadCredentialsException( //
41              "The presented AuthenticationToken does not contain the expected principal or credentials");
42          }
43  
44          if (!GenericUser.isGenericUserLogin((GenericUser) principal)) {
45            if (log.isDebugEnabled()) {
46              log.debug("not login");
47            }
48            throw new BadCredentialsException("login needed");
49          }
50  
51          result = AuthenticationTokens.authenticated(token);
52        }
53      } else {
54        result = null;
55      }
56  
57      return result;
58    }
59  
60    @Override
61    public boolean supports(final Class<?> authentication) {
62      return PreAuthenticatedAuthenticationToken.class.isAssignableFrom(authentication);
63    }
64  }
