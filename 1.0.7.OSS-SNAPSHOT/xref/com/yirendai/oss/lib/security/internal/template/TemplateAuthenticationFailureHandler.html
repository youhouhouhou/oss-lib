29  @Slf4j
30  public class TemplateAuthenticationFailureHandler implements AuthenticationFailureHandler {
31    // extends SimpleUrlAuthenticationFailureHandler
32  
33    private String defaultFailureUrl;
34    private boolean forwardToDestination = false;
35    private boolean allowSessionCreation = true;
36    private RedirectStrategy redirectStrategy = new DefaultRedirectStrategy();
37  
38    private ExceptionResolver<Throwable> exceptionResolver;
39    private TypeSafeCookie<ResolvedError> resolvedErrorCookie;
40  
41    public TemplateAuthenticationFailureHandler( //
42      final String defaultFailureUrl, //
43      final ExceptionResolver<Throwable> exceptionResolver, //
44      final TypeSafeCookie<ResolvedError> resolvedErrorCookie //
45    ) {
46      setDefaultFailureUrl(defaultFailureUrl);
47      this.exceptionResolver = exceptionResolver;
48      this.resolvedErrorCookie = resolvedErrorCookie;
49    }
50  
51    @Override
52    public void onAuthenticationFailure( //
53      final HttpServletRequest request, //
54      final HttpServletResponse response, //
55      final AuthenticationException exception //
56    ) throws IOException, ServletException {
57      final ResolvedError resolvedError = this.exceptionResolver.resolve(request, exception);
58      if (this.resolvedErrorCookie != null) {
59        this.resolvedErrorCookie.setCookie(request, response, resolvedError.eraseTraces());
60      }
61  
62      if (defaultFailureUrl == null) {
63        log.debug("No failure URL set, sending 401 Unauthorized error");
64  
65        response.sendError(HttpServletResponse.SC_UNAUTHORIZED,
66          "Authentication Failed: " + exception.getMessage());
67      } else {
68        saveException(request, exception);
69  
70        if (forwardToDestination) {
71          log.debug("Forwarding to " + defaultFailureUrl);
72  
73          request.getRequestDispatcher(defaultFailureUrl)
74            .forward(request, response);
75        } else {
76          final String url = this.defaultFailureUrl + "?error=" + urlEncode(resolvedError.getLocalizedMessage());
77          log.debug("Redirecting to " + url);
78          redirectStrategy.sendRedirect(request, response, url);
79        }
80      }
81    }
82  
83    /**
84     * <p>
85     * Caches the {@code AuthenticationException} for use in view rendering.
86     * </p>
87     * If {@code forwardToDestination} is set to true, request scope will be used,
88     * otherwise it will attempt to store the exception in the session. If there is no
89     * session and {@code allowSessionCreation} is {@code true} a session will be created.
90     * Otherwise the exception will not be stored.
91     *
92     * @param request   request
93     * @param exception exception
94     */
95    protected final void saveException(final HttpServletRequest request, final AuthenticationException exception) {
96      if (forwardToDestination) {
97        request.setAttribute(WebAttributes.AUTHENTICATION_EXCEPTION, exception);
98      } else {
99        HttpSession session = request.getSession(false);
100 
101       if (session != null || allowSessionCreation) {
102         request.getSession().setAttribute(WebAttributes.AUTHENTICATION_EXCEPTION, exception);
103       }
104     }
105   }

107   /**
108    * The URL which will be used as the failure destination.
109    *
110    * @param defaultFailureUrl the failure URL, for example "/loginFailed.jsp".
111    */
112   public void setDefaultFailureUrl(String defaultFailureUrl) {
113     Assert.isTrue(UrlUtils.isValidRedirectUrl(defaultFailureUrl), "'"
114       + defaultFailureUrl + "' is not a valid redirect URL");
115     this.defaultFailureUrl = defaultFailureUrl;
116   }

153   public static TemplateAuthenticationFailureHandler templateFailureHandler( //
154     final String loginFormUrl, //
155     final ExceptionResolver<Throwable> exceptionResolver, //
156     final TypeSafeCookie<ResolvedError> resolvedErrorCookie //
157   ) {
158     final TemplateAuthenticationFailureHandler failureHandler = new TemplateAuthenticationFailureHandler( //
159       loginFormUrl, exceptionResolver, resolvedErrorCookie);
160     failureHandler.setUseForward(false);
161     failureHandler.setRedirectStrategy(new SmartRedirectStrategy());
162     return failureHandler;
