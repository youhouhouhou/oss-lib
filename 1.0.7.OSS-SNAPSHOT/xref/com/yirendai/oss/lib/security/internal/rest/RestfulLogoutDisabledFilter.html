22  /**
23   * Note: Not a bean, avoid auto pick-up.
24   * see: {@link org.springframework.security.web.authentication.logout.LogoutFilter}.
25   * Created by zhanghaolun on 16/8/21.
26   */
27  public class RestfulLogoutDisabledFilter extends GenericFilterBean {
28  
29    private RequestMatcher logoutRequestMatcher;
30  
31    @Autowired
32    @NonNull
33    private RestfulExceptionHandler exceptionHandler;
34  
35    @Override
36    public void doFilter( //
37        final ServletRequest req, //
38        final ServletResponse res, //
39        final FilterChain chain //
40    ) throws IOException, ServletException {
41      final HttpServletRequest request = (HttpServletRequest) req;
42      final HttpServletResponse response = (HttpServletResponse) res;
43  
44      if (requiresLogout(request, response)) {
45        this.exceptionHandler.resolveAndHandle(request, response, new IllegalStateException("security disabled"));
46      } else {
47        chain.doFilter(request, response);
48      }
49    }
50  
51    /**
52     * Allow subclasses to modify when a logout should take place.
53     *
54     * @param request  the request
55     * @param response the response
56     * @return <code>true</code> if logout should occur, <code>false</code> otherwise
57     */
58    protected boolean requiresLogout( //
59        final HttpServletRequest request, //
60        final HttpServletResponse response //
61    ) {
62      return this.logoutRequestMatcher.matches(request);
63    }
64  
65    public void setLogoutRequestMatcher(final RequestMatcher logoutRequestMatcher) {
66      Assert.notNull(logoutRequestMatcher, "logoutRequestMatcher cannot be null");
67      this.logoutRequestMatcher = logoutRequestMatcher;
68    }
69  
70    public void setFilterProcessesUrl(final String filterProcessesUrl) {
71      this.logoutRequestMatcher = new AntPathRequestMatcher(filterProcessesUrl);
72    }
73  
74    public void setExceptionHandler(final RestfulExceptionHandler exceptionHandler) {
75      this.exceptionHandler = exceptionHandler;
76    }
77  }
