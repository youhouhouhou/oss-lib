24  @AutoConfigureAfter({WebApplicationSecurityAutoConfiguration.class})
25  @ConditionalOnAppSecurity(ENABLED)
26  @ConditionalOnClass({HasFeatures.class, NamedContextFactory.class, ZuulFilter.class})
27  @Configuration
28  public class ZuulTokenConfiguration {
29  
30    @Qualifier(GENERIC_USER_TOKEN)
31    @Autowired
32    private TypeSafeToken<GenericUser> genericUserToken;
33  
34    @Bean
35    public ContextAuthTokenZuulFilter contextAuthTokenZuulFilter() {
36      final ContextAuthTokenZuulFilter filter = new ContextAuthTokenZuulFilter();
37      filter.setToken(this.genericUserToken);
38      return filter;
39    }
40  }
