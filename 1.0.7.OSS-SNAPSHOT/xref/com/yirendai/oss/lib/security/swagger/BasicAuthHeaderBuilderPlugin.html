27  @Order(SwaggerPluginSupport.SWAGGER_PLUGIN_ORDER)
28  public class BasicAuthHeaderBuilderPlugin implements OperationBuilderPlugin {
29  
30    private final ParameterBuilder parameterBuilder = new ParameterBuilder();
31  
32    @Autowired
33    private AppProperties appProperties;
34    @Autowired(required = false)
35    private SecurityProperties securityProperties;
36  
37    @Override
38    public void apply(final OperationContext operationContext) {
39      final Boolean basicEnabled = this.securityProperties != null && this.securityProperties.getBasic().isEnabled();
40      final AppSecurityProperties appSecurityProperties = this.appProperties.getSecurity();
41  
42      final String loginEndpoint = appSecurityProperties.getLoginProcessingUrl();
43      // Get endpoint request mapping
44      final String mapping = operationContext.requestMappingPattern();
45  
46      // Create auth header parameter
47      if ((basicEnabled || appSecurityProperties.getEnabled()) && !appProperties.getType().equals(AppType.RESOURCE)
48          && !appProperties.getType().equals(AppType.TEMPLATE) && !loginEndpoint.equals(mapping)) {
49        final String description = "Authorization: Basic <_your_token_>, " + //
50          "echo \"Basic $(echo -ne \\\"$username:$password\\\" | base64)\"";
51        final List<Parameter> parameters = Lists.newArrayList(
52          this.parameterBuilder
53            .parameterType("header")
54            .name(BasicAuthUtils.BASIC_AUTH_HEADE_NAME)
55            .modelRef(new ModelRef("string"))
56            .defaultValue("Basic ")
57            .description(description)
58            .allowMultiple(false)
59            .required(false)
60            .build() //
61        );
62  
63        // Add parameter to endpoint documentation
64        operationContext.operationBuilder().parameters(parameters);
65      }
66    }
67  
68    @Override
69    public boolean supports(final DocumentationType delimiter) {
70      return DocumentationType.SWAGGER_2.equals(delimiter);
71    }
72  }
