38  @Order(SwaggerPluginSupport.SWAGGER_PLUGIN_ORDER + 10)
39  @Slf4j
40  public class AfterOperationBuilderBuildPlugin implements OperationBuilderPlugin {
41  
42    private static final String MANUAL_REQUEST_HANDLER = ManualRequestHandler.class.getSimpleName();
43  
44    @Autowired
45    private TypeResolver typeResolver;
46    @Autowired
47    private TypeNameExtractor nameExtractor;
48  
49    @Override
50    public void apply(final OperationContext context) {
51      try {
52        // 处理自定义的请求处理器
53        if (MANUAL_REQUEST_HANDLER.equals(context.getName())) {
54  
55          // get requestContext
56          Field privateStringField = OperationContext.class.getDeclaredField("requestContext");
57          privateStringField.setAccessible(true);
58          final RequestMappingContext requestMappingContext = //
59            (RequestMappingContext) privateStringField.get(context);
60  
61          // get handler
62          privateStringField = RequestMappingContext.class.getDeclaredField("handler");
63          privateStringField.setAccessible(true);
64          final RequestHandler requestHandler = //
65            (RequestHandler) privateStringField.get(requestMappingContext);
66  
67          if (requestHandler != null && requestHandler instanceof ManualRequestHandler) {
68            final ManualRequestHandler manualRequestHandler = (ManualRequestHandler) requestHandler;
69            final ApiOperationInfo apiOperationInfo = manualRequestHandler.getApiOperationInfo();
70  
71            if (apiOperationInfo != null) {
72              // 接口描述
73              context.operationBuilder().notes(apiOperationInfo.getNotes());
74              // 请求参数处理
75              if (apiOperationInfo.getApiRequest().getParameters() != null) {
76                context.operationBuilder().parameters(apiOperationInfo.getApiRequest().getParameters());
77              }
78            }
79  
80          }
81        }
82  
83        // 处理响应的错误码对应的ResolveError
84        processResponseModel(context);
85      } catch (final ReflectiveOperationException ex) {
86        log.error("swagger plugin error.", ex);
87      }
88    }
89  
90    /**
91     * 为错误码的响应增加Model的映射 @com.yirendai.oss.lib.errorhandle.api.ResolvedError
92     */
93    private void processResponseModel(OperationContext context) {
94      try {
95        final OperationBuilder operationBuilder = context.operationBuilder();
96        final Field privateStringField = OperationBuilder.class.getDeclaredField("responseMessages");
97        privateStringField.setAccessible(true);
98        final Set<ResponseMessage> responseMessages = (Set<ResponseMessage>) privateStringField.get(operationBuilder);
99  
100       final ModelContext modelContext = returnValue( //
101         this.typeResolver.resolve(context.getReturnType()),
102         context.getDocumentationType(),
103         context.getAlternateTypeProvider(),
104         context.getGenericsNamingStrategy(),
105         context.getIgnorableParameterTypes());
106       final ModelReference responseModel = modelRefFactory(modelContext, this.nameExtractor) //
107         .apply(context.alternateFor(this.typeResolver.resolve(ResolvedError.class)));
108 
109       final Set<ResponseMessage> responseMessagesProcessed = newHashSet();
110       responseMessages.stream().forEach((responseMessage -> {
111         if (!isSuccessful(responseMessage.getCode())) {
112           responseMessagesProcessed.add(new ResponseMessageBuilder()
113             .code(responseMessage.getCode())
114             .message(responseMessage.getMessage())
115             .responseModel(responseModel)
116             .build());
117         }
118       }));
119       if (!responseMessagesProcessed.isEmpty()) {
120         context.operationBuilder().responseMessages(responseMessagesProcessed);
121       }
122     } catch (final NoSuchFieldException | IllegalAccessException ex) {
123       log.info("error processResponseModel.", ex);
124     }
125   }
126 
127   /**
128    * 判断是否是成功的响应.
129    */
130   private static boolean isSuccessful(final int code) {
131     try {
132       return HttpStatus.Series.SUCCESSFUL.equals(HttpStatus.Series.valueOf(code));
133     } catch (final Exception ignored) {
134       log.info("error check httpStatus {}.", code, ignored);
135       return false;
136     }
137   }
138 
139   @Override
140   public boolean supports(final DocumentationType delimiter) {
141     return DocumentationType.SWAGGER_2.equals(delimiter);
142   }
