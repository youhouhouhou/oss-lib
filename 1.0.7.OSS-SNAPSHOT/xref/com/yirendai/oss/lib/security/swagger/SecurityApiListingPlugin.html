49      //ApiListingBuilderPlugin, this will override all api
50      //  apiListingContext.apiListingBuilder() //
51      //  .apis(this.additionalOperations());
52  
53      final Class<?> controllerClass = apiListingContext.getResourceGroup().getControllerClass();
54      if (controllerClass != null && !controllerClass.getName().startsWith("org.springframework.boot.actuate")) {
55        final ApiListingBuilder builder = apiListingContext.apiListingBuilder();
56        final Field field = ReflectionUtils.findField(builder.getClass(), "apis");
57        ReflectionUtils.makeAccessible(field);
58        final Object value = Optional.ofNullable(ReflectionUtils.getField(field, builder)) //
59          .orElse(Collections.emptyList());
60        @SuppressWarnings("unchecked")
61        final List<ApiDescription> apis = (List<ApiDescription>) value;
62  
63        final List<ApiDescription> newApis = new ArrayList<>(apis);
64        newApis.addAll(this.additionalOperations());
65  
66        builder.apis(newApis);
67      }
68    }
