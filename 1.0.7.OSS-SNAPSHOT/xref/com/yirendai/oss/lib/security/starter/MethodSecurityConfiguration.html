9   import org.springframework.beans.factory.annotation.Autowired;
10  import org.springframework.boot.autoconfigure.condition.ConditionalOnBean;
11  import org.springframework.security.access.PermissionEvaluator;
12  import org.springframework.security.access.expression.DenyAllPermissionEvaluator;
13  import org.springframework.security.access.expression.method.DefaultMethodSecurityExpressionHandler;
14  import org.springframework.security.access.expression.method.MethodSecurityExpressionHandler;
15  import org.springframework.security.config.annotation.method.configuration.EnableGlobalMethodSecurity;
16  import org.springframework.security.config.annotation.method.configuration.GlobalMethodSecurityConfiguration;
17  
18  /**
19   * Created by zhanghaolun on 16/7/13.
20   */
21  @EnableGlobalMethodSecurity(jsr250Enabled = true, securedEnabled = true, prePostEnabled = true)
22  @ConditionalOnAppSecurity(ENABLED)
23  @ConditionalOnBean(value = {PermissionEvaluator.class})
24  public class MethodSecurityConfiguration extends GlobalMethodSecurityConfiguration {
25  
26    @Autowired
27    private AppProperties appProperties;
28  
29    @Autowired(required = false)
30    private PermissionEvaluator projectPermissionEvaluator;
31  
32    @Override
33    protected MethodSecurityExpressionHandler createExpressionHandler() {
34      final PermissionEvaluator permissionEvaluator;
35      if (this.projectPermissionEvaluator != null) {
36        final MethodSecurityPermissionEvaluator methodSecurityPermissionEvaluator =
37          new MethodSecurityPermissionEvaluator();
38        methodSecurityPermissionEvaluator.setSecurityEnabled(this.appProperties.getSecurityEnabled());
39        methodSecurityPermissionEvaluator.setProjectPermissionEvaluator(this.projectPermissionEvaluator);
40        permissionEvaluator = methodSecurityPermissionEvaluator;
41      } else {
42        permissionEvaluator = new DenyAllPermissionEvaluator();
43      }
44  
45      final DefaultMethodSecurityExpressionHandler expressionHandler = new DefaultMethodSecurityExpressionHandler();
46      expressionHandler.setPermissionEvaluator(permissionEvaluator);
47      return expressionHandler;
48    }
49  }
