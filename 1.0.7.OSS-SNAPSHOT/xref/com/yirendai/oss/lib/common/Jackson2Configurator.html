36    default Optional<Class<?>> findClass(final String className) {
37      Class<?> classFound;
38      try {
39        classFound = Class.forName(className);
40      } catch (final ClassNotFoundException ex) {
41        LoggerFactory.getLogger(Jackson2Configurator.class).debug("{} not found", className, ex);
42        classFound = null;
43      }
44      return Optional.ofNullable(classFound);
45    }
46  
47    default Optional<String> getProperty(final PropertyResolver propertyResolver, final String key) {
48      final Optional<String> result;
49      if (propertyResolver != null) {
50        final String property = propertyResolver.getProperty(key);
51        result = Optional.ofNullable(property);
52      } else {
53        result = Optional.empty();
54      }
55      return result;
56    }
57  
58    default <M extends ObjectMapper> Boolean isXmlMapper(final M mapper) {
59      final Optional<Class<?>> optional = findClass(XMLMAPPER_CLASSNAME);
60      return optional.map(xmlMapperClass -> xmlMapperClass.isAssignableFrom(mapper.getClass())).orElse(FALSE);
61    }

80          mapper.configure(JsonGenerator.Feature.QUOTE_NON_NUMERIC_NUMBERS, false);
81          mapper.configure(JsonParser.Feature.ALLOW_NON_NUMERIC_NUMBERS, true);
82          return mapper;

86        @Override
87        public <M extends ObjectMapper> M config(final PropertyResolver propertyResolver, final M mapper) {
88          mapper.setTimeZone(Defaults.UTC_P8.toTimeZone());
89          mapper.registerModule(new JodaModule());
90          // Jdk8Module ?
91          mapper.disable(WRITE_DATES_AS_TIMESTAMPS);

95          formatJdk.setTimeZone(UTC_P8.toTimeZone());
96          mapper.setDateFormat(formatJdk);
97          return mapper;

100     JACKSON2_HAL_CONFIGURATOR {
101 
102       private static final String MODULE_CLASS = "org.springframework.hateoas.hal.Jackson2HalModule";
103 
104       @Override
105       public <M extends ObjectMapper> M config(final PropertyResolver propertyResolver, final M mapper) {
106         final Optional<Class<?>> moduleClass = this.findClass(MODULE_CLASS);
107         if (moduleClass.isPresent()) {
108           // need HalHandlerInstantiator or lead to exception on data-rest request
109           final Boolean isAlreadyRegisteredIn = this.isAlreadyRegisteredIn(mapper, moduleClass.get());
110           if (!isAlreadyRegisteredIn) {
111             try {
112               mapper.registerModule((Module) moduleClass.get().newInstance());
113             } catch (final ReflectiveOperationException ex) {
114               log.info("Jackson2HalModule config error", ex);
115             }
116           }
117         }
118         return mapper;
119       }
120 
121       <M extends ObjectMapper> Boolean isAlreadyRegisteredIn(final M mapper, final Class<?> jackson2HalModuleClass) {
122         Boolean result;
123         try {
124           final Method isAlreadyRegisteredIn = jackson2HalModuleClass.getDeclaredMethod( //
125             "isAlreadyRegisteredIn", ObjectMapper.class);
126           result = (Boolean) isAlreadyRegisteredIn.invoke(null, mapper);
127         } catch (final ReflectiveOperationException | SecurityException | IllegalArgumentException ex) {
128           log.info("Jackson2HalModule config error", ex);
129           result = FALSE;
130         }
131         return result;
132       }

138     JACKSON2_JAXB_ANNOTATION_CONFIGUATOR {
139 
140       private static final String MODULE_CLASS = "com.fasterxml.jackson.module.jaxb.JaxbAnnotationModule";
141       private static final String JAXB_ANNOTATION_INTROSPECTOR_CLASS = //
142         "com.fasterxml.jackson.module.jaxb.JaxbAnnotationIntrospector";
143 
144       @Override
145       public <M extends ObjectMapper> M config(final PropertyResolver propertyResolver, final M mapper) {
146         final Boolean isXmlMapper = this.isXmlMapper(mapper);
147         final String jaxbEnabledDefault = isXmlMapper ? "true" : "false";
148         final Boolean jaxbEnabled = parseBoolean( //
149           this.getProperty(propertyResolver, JACKSON_JAXB_ENABLED).orElse(jaxbEnabledDefault));
150 
151         final Optional<Class<?>> moduleClass = this.findClass(MODULE_CLASS);
152         if (jaxbEnabled && moduleClass.isPresent()) {
153           try {
154             final Class<?> jaxbClass = this.findClass(JAXB_ANNOTATION_INTROSPECTOR_CLASS).get();
155             final Module module = (Module) moduleClass.get().getConstructor(jaxbClass) //
156               .newInstance(new Jackson2HackedJaxbAnnotationIntrospector());
157             final Class<?> enumType = this.findClass(MODULE_CLASS + "$Priority").orElse(null);
158 
159             final String priorityName = isXmlMapper ? "PRIMARY" : "SECONDARY";
160             final Object priority = this.getEnumValue(enumType, priorityName);
161             final Method setPriorityMethod = moduleClass.get().getDeclaredMethod("setPriority", enumType);
162             setPriorityMethod.invoke(module, priority);
163 
164             mapper.registerModule(module);
165           } catch (final ReflectiveOperationException ex) {
166             log.info("JaxbAnnotationModule config error", ex);
167           }
168         }
169         return mapper;
170       }
171 
172       private Object getEnumValue(final Class<?> enumType, final String name) throws ReflectiveOperationException {
173         final Object result;
174         if (enumType != null) {
175           final Method method = enumType.getDeclaredMethod("valueOf", String.class);
176           result = method.invoke(enumType, name);
177         } else {
178           result = null;
179         }
180         return result;
181       }
