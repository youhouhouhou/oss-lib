23  public class AesDecryptor implements EncodeDecryptor {
24  
25    @Getter
26    private final Charset charset;
27    private final Provider provider;
28    private final AesCbcKey key;
29  
30    public AesDecryptor(final Provider provider, final KeyExpression keyExpression) {
31      this.charset = UTF_8;
32      this.provider = provider;
33      this.key = new AesCbcKey(keyExpression);
34    }
35  
36    @Override
37    public String decrypt(final String encryptedAndEncoded) {
38      return new String(this.decryptBytes(encryptedAndEncoded), this.charset);
39    }
40  
41    public byte[] decryptBytes(final String encryptedAndEncoded) {
42      final byte[] unencrypted;
43      try {
44        // 使用BASE64对密文进行解码
45        final byte[] decoded = CodecUtils.decodeBase64(encryptedAndEncoded);
46        // 解密
47        unencrypted = this.decryptCipher().doFinal(decoded);
48      } catch (final IllegalBlockSizeException | BadPaddingException | IllegalArgumentException cause) {
49        throw new AesException(AesException.AesError.DECRYPT_AES_ERROR, cause);
50      }
51      try {
52        // 去除补位字符
53        final byte[] unpad = Pkcs7Encoder.PKCS7_UTF8_BLOCK32.decode(unencrypted);
54        return copyOfRange(unpad, 16, unpad.length);
55      } catch (final IllegalArgumentException cause) {
56        throw new AesException(AesException.AesError.ILLEGAL_BUFFER, cause);
57      }
58    }
59  
60    @SneakyThrows
61    Cipher decryptCipher() {
62      final Cipher cipher = Cipher.getInstance(AES_CBC_NOPADDING, this.provider);// 设置解密模式为AES的CBC模式
63      cipher.init(DECRYPT_MODE, this.key.getKeySpec(), this.key.getCbcIv());
64      return cipher;
65    }
66  }
