58    public RsaKeyGenerator(final String spec) {
59      this.provider = Cryptos.provider();
60      this.spec = spec;
61      this.keyFormat = keyFormat(spec);
62      this.keySize = RsaKey.keySize(spec);
63      this.keyType = RsaKey.keyType(spec);
64      checkArgument(SUPPORTED_PAIR_FORMATS.contains(this.keyFormat), "keyFormat " + this.keyFormat + " not supported.");
65      checkArgument(KEY_TYPE_PAIR.equals(this.keyType), "keyType " + this.keyType + " not supported.");
66    }

68    @Override
69    public KeyExpression generateKey() {
70      final KeyExpression keyExpression = RsaKeyGenerator.generateRsaKeyPair( //
71        this.provider, this.keyFormat, this.keySize);
72      this.key = new RsaKey(keyExpression);
73      return this.key.getKeyExpression();
74    }
75  
76    @Override
77    public KeyExpression getKey(final String spec) {
78      if (this.key == null) {
79        this.generateKey();
80      }
81  
82      return this.key.getKey(spec);
83    }

85    public static KeyExpression convertPairFromPkcs8X509ToPkcs1(final KeyExpression pairPkcs8X509) {
86      checkArgument(KEY_FORMAT_PKCS8_X509.equals(keyFormat(pairPkcs8X509.getSpec())), //
87        "unsupported spec" + pairPkcs8X509.getSpec());
88      final byte[] privateKeyPkcs8 = CodecUtils.decodeBase64(extractPrivateKey(pairPkcs8X509));
89      final byte[] publicKeyX509 = CodecUtils.decodeBase64(extractPublicKey(pairPkcs8X509));
90      final String privateKeyPem = convertPrivateKeyFromPkcs8ToPkcs1Pem(privateKeyPkcs8);
91      final String publicKeyPem = convertPublicKeyFromX509ToPkcs1Pem(publicKeyX509);
92  
93      final int keySize = keySize(pairPkcs8X509.getSpec());
94      final String spec = RsaKey.keySpec(KEY_FORMAT_PKCS1 + UNDERSCORE + KEY_FORMAT_PKCS1, keySize, KEY_TYPE_PAIR);
95      final String value = dropComment(privateKeyPem, COMMENT_MARK) + COLON + dropComment(publicKeyPem, COMMENT_MARK);
96      return new KeyExpression(spec, value);
97    }

99    @SneakyThrows
100   public static String convertPrivateKeyFromPkcs8ToPkcs1Pem(final byte[] privateKeyPkcs8) {
101     // Convert private key from PKCS8 to PKCS1:
102     final ASN1Encodable encodable = PrivateKeyInfo.getInstance(privateKeyPkcs8).parsePrivateKey();
103     return pem(encodable.toASN1Primitive().getEncoded(), KEY_FORMAT_PKCS1, KEY_TYPE_PRIVATE);
104   }
105 
106   @SneakyThrows
107   public static String convertPublicKeyFromX509ToPkcs1Pem(final byte[] publicKeyX509) {
108     // Convert public key from X.509 SubjectPublicKeyInfo to PKCS1:
109     final ASN1Primitive publicKeyPrimitive = SubjectPublicKeyInfo.getInstance(publicKeyX509).parsePublicKey();
110     return pem(publicKeyPrimitive.getEncoded(), KEY_FORMAT_PKCS1, KEY_TYPE_PUBLIC);
111   }

113   public static KeyExpression generateRsaKeyPair( //
114     final Provider provider, final String keyFormat, final int keySize //
115   ) {
116     final String privateKeyFormat = keyFormat.split(UNDERSCORE)[0];
117     final String publicKeyFormat = keyFormat.split(UNDERSCORE)[1];
118 
119     final KeyExpression pairPkcs8X509 = generateRsaKeyPairPkcs8X509(provider, keySize);
120     final KeyExpression pairPkcs1 = convertPairFromPkcs8X509ToPkcs1(pairPkcs8X509);
121 
122     final String privateKey;
123     if (KEY_FORMAT_PKCS8.equals(privateKeyFormat)) {
124       privateKey = RsaKey.extractPrivateKey(pairPkcs8X509);
125     } else if (KEY_FORMAT_PKCS1.equals(privateKeyFormat)) {
126       privateKey = RsaKey.extractPrivateKey(pairPkcs1);
127     } else {
128       throw new IllegalArgumentException("unsupported privateKeyFormat " + privateKeyFormat);
129     }
130 
131     final String publicKey;
132     if (KEY_FORMAT_X509.equals(publicKeyFormat)) {
133       publicKey = RsaKey.extractPublicKey(pairPkcs8X509);
134     } else if (KEY_FORMAT_PKCS1.equals(publicKeyFormat)) {
135       publicKey = RsaKey.extractPublicKey(pairPkcs1);
136     } else {
137       throw new IllegalArgumentException("unsupported publicKeyFormat " + publicKeyFormat);
138     }
139 
140     final String spec = RsaKey.keySpec(privateKeyFormat + UNDERSCORE + publicKeyFormat, keySize, KEY_TYPE_PAIR);
141     final String value = privateKey + COLON + publicKey;
142     return new KeyExpression(spec, value);
143   }

145   @SneakyThrows
146   public static KeyPair generateRsaKeyPair(final Provider provider, final int keySize) {
147     final KeyPairGenerator keyPairGen = KeyPairGenerator.getInstance(ALGO_RSA, provider);
148     // final byte[] seed = DateFormatUtils.format(currentTimeMillis(), "yyyyMMdd").getBytes();
149     keyPairGen.initialize(keySize, new SecureRandom());
150     //keygen.initialize(keySize);
151     return keyPairGen.generateKeyPair();
152   }

154   @SneakyThrows
155   public static KeyExpression generateRsaKeyPairPkcs8X509(
156     final Provider provider, //
157     final int keySize //
158   ) {
159     final KeyPair pairPkcs8X509 = generateRsaKeyPair(provider, keySize);
160     final byte[] privateKeyPkcs8 = pairPkcs8X509.getPrivate().getEncoded();
161     final byte[] publicKeyX509 = pairPkcs8X509.getPublic().getEncoded();
162 
163     final String privateKeyPem = pem(privateKeyPkcs8, KEY_FORMAT_PKCS8, KEY_TYPE_PRIVATE);
164     final String publicKeyPem = pem(publicKeyX509, KEY_FORMAT_X509, KEY_TYPE_PUBLIC);
165 
166     final String spec = RsaKey.keySpec(KEY_FORMAT_PKCS8 + UNDERSCORE + KEY_FORMAT_X509, keySize, KEY_TYPE_PAIR);
167     final String value = dropComment(privateKeyPem, COMMENT_MARK) + COLON + dropComment(publicKeyPem, COMMENT_MARK);
168     return new KeyExpression(spec, value);
169   }

171   @SneakyThrows
172   public static String pem(final byte[] bytes, final String keyFormat, final String keyType) {
173     final String type;
174     if (KEY_TYPE_PRIVATE.equals(keyType)) {
175       type = "RSA PRIVATE KEY";
176     } else if (KEY_TYPE_PUBLIC.equals(keyType)) {
177       if (KEY_FORMAT_X509.equals(keyFormat)) {
178         type = "PUBLIC KEY";
179       } else {
180         type = "RSA PUBLIC KEY";
181       }
182     } else {
183       throw new IllegalArgumentException("unsupported keyType " + keyType);
184     }
185 
186     final StringWriter stringWriter = new StringWriter();
187     try (final PemWriter pemWriter = new PemWriter(stringWriter)) {
188       pemWriter.writeObject(new PemObject(type, bytes));
189     }
190     return stringWriter.toString();
191   }
