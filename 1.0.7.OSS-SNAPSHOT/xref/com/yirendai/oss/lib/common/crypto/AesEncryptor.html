22  public class AesEncryptor implements EncodeEncryptor {
23  
24    @Getter
25    private final Charset charset;
26    private final Provider provider;
27    private final AesCbcKey key;
28  
29    public AesEncryptor(final Provider provider, final KeyExpression keyExpression) {
30      this.charset = UTF_8;
31      this.provider = provider;
32      this.key = new AesCbcKey(keyExpression);
33    }
34  
35    @Override
36    public String encrypt(final String plainText) {
37      return this.encryptBytes( //
38        RandomStrings.RANDOM_BASE62.generate(this.key.getCbcIvLength()), //
39        plainText.getBytes(this.charset) //
40      );
41    }
42  
43    @Override
44    public String encrypt(final String plainText, final Integer maxAge) {
45      throw new UnsupportedOperationException();
46    }
47  
48    public String encryptBytes(final String random, final byte[] plainBytes) {
49      final byte[] randomBytes = random.getBytes(this.charset);
50      final int nopadLength = randomBytes.length + plainBytes.length;
51      // ... + pad: 使用自定义的填充方式对明文进行补位填充
52      final byte[] padBytes = Pkcs7Encoder.PKCS7_UTF8_BLOCK32.encode(nopadLength);
53  
54      // 获得最终的字节流, 未加密
55      // random + plain + pad
56      final ByteBuffer unencrypted = ByteBuffer.allocate(nopadLength + padBytes.length);
57      unencrypted.put(randomBytes);
58      unencrypted.put(plainBytes);
59      unencrypted.put(padBytes);
60      try {
61        // 加密
62        final byte[] encrypted = this.encryptCipher().doFinal(unencrypted.array());
63        // 使用BASE64对加密后的字符串进行编码
64        return CodecUtils.encodeBase64(encrypted);
65      } catch (final Exception cause) {
66        throw new AesException(AesException.AesError.ENCRYPT_AES_ERROR, cause);
67      } finally {
68        unencrypted.clear();
69      }
70    }
71  
72    @SneakyThrows
73    Cipher encryptCipher() { // arg1: "BC" 指定使用 bouncycastle provider, 不指定则使用jre
74      final Cipher cipher = Cipher.getInstance(AES_CBC_NOPADDING, this.provider);// 设置加密模式为AES的CBC模式
75      cipher.init(ENCRYPT_MODE, this.key.getKeySpec(), this.key.getCbcIv());
76      return cipher;
77    }
78  }
