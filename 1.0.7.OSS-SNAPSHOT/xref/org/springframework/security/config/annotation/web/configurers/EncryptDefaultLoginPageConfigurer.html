29  public class EncryptDefaultLoginPageConfigurer<H extends HttpSecurityBuilder<H>> extends
30    AbstractHttpConfigurer<DefaultLoginPageConfigurer<H>, H> {
31  
32    private String loginPage;
33    private DefaultLoginPageGeneratingFilter loginPageGeneratingFilter;
34    private Boolean useDefaultLoginPage;
35    private ExceptionResolver<Throwable> exceptionResolver;
36    private TypeSafeCookie<ResolvedError> resolvedErrorCookie;
37  
38    public EncryptDefaultLoginPageConfigurer( //
39      final String loginPage, //
40      final String publicKey, //
41      final Boolean useDefaultLoginPage, //
42      final ExceptionResolver<Throwable> exceptionResolver, //
43      final TypeSafeCookie<ResolvedError> resolvedErrorCookie //
44    ) {
45      final EncryptDefaultLoginPageGeneratingFilter filter = new EncryptDefaultLoginPageGeneratingFilter();
46      filter.setRsaPublicKey(publicKey);
47      filter.setLoginPageUrl(loginPage);
48      this.loginPage = loginPage;
49      this.loginPageGeneratingFilter = filter;
50      this.useDefaultLoginPage = useDefaultLoginPage;
51  
52      this.exceptionResolver = exceptionResolver;
53      this.resolvedErrorCookie = resolvedErrorCookie;
54    }
55  
56    @Override
57    public void init(final H http) throws Exception {
58      http.setSharedObject(DefaultLoginPageGeneratingFilter.class, this.loginPageGeneratingFilter);
59    }
60  
61    @Override
62    @SuppressWarnings("unchecked")
63    public void configure(final H http) throws Exception {
64      final ExceptionHandlingConfigurer<?> exceptionConf = http.getConfigurer(ExceptionHandlingConfigurer.class);
65      //final AuthenticationEntryPoint authenticationEntryPoint = exceptionConf != null ?
66      // exceptionConf.getAuthenticationEntryPoint() : null;
67  
68      if (this.loginPageGeneratingFilter.isEnabled() && this.useDefaultLoginPage) { // authenticationEntryPoint == null
69        this.loginPageGeneratingFilter.setLoginPageUrl(this.loginPage);
70        this.configure(exceptionConf);
71        this.loginPageGeneratingFilter = postProcess(this.loginPageGeneratingFilter);
72        http.addFilter(this.loginPageGeneratingFilter);
73      }
74    }
75  
76    @SuppressWarnings("unchecked")
77    private void configure(final ExceptionHandlingConfigurer<?> exceptionHandling) {
78      if (exceptionHandling == null) {
79        return;
80      }
81      // using DelegatingAuthenticationEntryPoint
82      // AbstractAuthenticationFilterConfigurer#registerDefaultAuthenticationEntryPoint()
83      final Field field = findField(ExceptionHandlingConfigurer.class, "defaultEntryPointMappings", LinkedHashMap.class);
84      makeAccessible(field);
85      final Map<RequestMatcher, AuthenticationEntryPoint> defaultEntryPointMappings = //
86        (Map<RequestMatcher, AuthenticationEntryPoint>) getField(field, exceptionHandling);
87  
88      for (final Entry<RequestMatcher, AuthenticationEntryPoint> entry : defaultEntryPointMappings.entrySet()) {
89        final AuthenticationEntryPoint value = entry.getValue();
90        if (isAssignableValue(LoginUrlAuthenticationEntryPoint.class, value)) {
91          ///final LoginUrlAuthenticationEntryPoint current = (LoginUrlAuthenticationEntryPoint) value;
92          final TemplateAuthenticationEntryPoint replacement = templateEntryPoint( //
93            this.loginPage, this.exceptionResolver, this.resolvedErrorCookie);
94          entry.setValue(replacement);
95        }
96      }
97    }
98  }
