28    private void testApplicationInSeparateClassLoader() throws Exception {
29      //run application code in separate class loader in order to isolate static state between test runs
30      final Runnable runnable = mock(Runnable.class);
31      //set up your mock object expectations here, if needed
32      InterfaceToApplicationDependentCode tester = makeCodeToRunInSeparateClassLoader(
33        "com.yirendai", InterfaceToApplicationDependentCode.class, CodeToRunInApplicationClassLoader.class);
34      //if you want to try the code without class loader isolation, comment out above line and comment in the line below
35      //CodeToRunInApplicationClassLoader tester = new CodeToRunInApplicationClassLoaderImpl();
36      tester.testTheCode(runnable);
37      verify(runnable).run();
38      assertEquals("should be one invocation!", 1, tester.getNumOfInvocations());
39    }

41    /**
42     * Create a new class loader for loading application-dependent code and return an instance of that.
43     */
44    @SuppressWarnings("unchecked")
45    private <I, T> I makeCodeToRunInSeparateClassLoader(
46      String packageName, Class<I> testCodeInterfaceClass, Class<T> testCodeImplClass) throws Exception {
47      TestApplicationClassLoader cl = new TestApplicationClassLoader(
48        packageName, getClass(), testCodeInterfaceClass);
49      Class<?> testerClass = cl.loadClass(testCodeImplClass.getName());
50      return (I) testerClass.newInstance();
51    }

53    /**
54     * Bridge interface, implemented by code that should be run in application class loader.
55     * This interface is loaded by the same class loader as the unit test class, so
56     * we can call the application-dependent code without need for reflection.
57     */
58    public interface InterfaceToApplicationDependentCode {
59      void testTheCode(Runnable run);
60      int getNumOfInvocations();
61    }
62  
63    /**
64     * Test-specific code to call application-dependent code. This class is loaded by
65     * the same class loader as the application code.
66     */
67    public static class CodeToRunInApplicationClassLoader implements InterfaceToApplicationDependentCode {
68      private static int numOfInvocations = 0;
69  
70      @Override
71      public void testTheCode(Runnable runnable) {
72        numOfInvocations++;
73        runnable.run();
74      }
75  
76      @Override
77      public int getNumOfInvocations() {
78        return numOfInvocations;
79      }
80    }

82    /**
83     * Loads application classes in separate class loader from test classes.
84     */
85    private static class TestApplicationClassLoader extends URLClassLoader {
86  
87      private final String appPackage;
88      private final String mainTestClassName;
89      private final String[] testSupportClassNames;
90  
91      public TestApplicationClassLoader(String appPackage, Class<?> mainTestClass, Class<?>... testSupportClasses) {
92        super(((URLClassLoader) getSystemClassLoader()).getURLs());
93        this.appPackage = appPackage;
94        this.mainTestClassName = mainTestClass.getName();
95        this.testSupportClassNames = convertClassesToStrings(testSupportClasses);
96      }
97  
98      private String[] convertClassesToStrings(Class<?>[] classes) {
99        String[] results = new String[classes.length];
100       for (int i = 0; i < classes.length; i++) {
101         results[i] = classes[i].getName();
102       }
103       return results;
104     }
105 
106     @Override
107     public Class<?> loadClass(String className) throws ClassNotFoundException {
108       if (isApplicationClass(className)) {
109         //look for class only in local class loader
110         return super.findClass(className);
111       }
112       //look for class in parent class loader first and only then in local class loader
113       return super.loadClass(className);
114     }
115 
116     private boolean isApplicationClass(String className) {
117       if (mainTestClassName.equals(className)) {
118         return false;
119       }
120       for (int i = 0; i < testSupportClassNames.length; i++) {
121         if (testSupportClassNames[i].equals(className)) {
122           return false;
123         }
124       }
125       return className.startsWith(appPackage);
126     }
